┃  BACKUP                                                        ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

    # Créer un backup
    cd /var/www/testing.mesnew.fr
    sudo tar -czf backup-$(date +%Y%m%d-%H%M%S).tar.gz *

    # Lister les backups
    ls -lht backup-*.tar.gz

    # Restaurer un backup
    sudo tar -xzf backup-YYYYMMDD-HHMMSS.tar.gz
